function goDownload() { const oid = document.getElementById('orderId').value.trim(); const msgDiv = document.getElementById('msg'); if(!oid) return; const salt = "EIVEN_SECURE_2026_SYSTEM_SALT_X9"; const hash = CryptoJS.SHA256(oid + salt).toString(CryptoJS.enc.Hex).toLowerCase().substring(0, 16); // 指向 Builder.py 自动生成的压缩包 const zipUrl = `./releases/${hash}/EIVEN_Package.zip`; msgDiv.style.color = "#d35400"; msgDiv.innerText = "正在检索您的专属加密压缩包..."; fetch(zipUrl, { method: 'HEAD' }) .then(res => { if(res.ok) { msgDiv.style.color = "#00ff00"; msgDiv.innerText = "✅ 验证通过！正在为您启动下载..."; window.location.href = zipUrl; // 触发 ZIP 下载 } else { msgDiv.style.color = "#ff4444"; msgDiv.innerText = "❌ 未找到文件。请确认验证码正确或联系管理员同步。"; } }) .catch(() => { msgDiv.innerText = "❌ 连接异常，请检查网络或代理设置。"; }); }
